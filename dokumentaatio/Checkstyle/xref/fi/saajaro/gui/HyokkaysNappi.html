10  import java.awt.event.ActionListener;
11  import java.util.ArrayList;
12  import javax.swing.JButton;
13  import javax.swing.JTextArea;
14  
15  public class HyokkaysNappi implements ActionListener {
16  
17      private String komento;
18      private Pelaaja hero;
19      private Mobs enemy;
20      private JTextArea kohde;
21      private Moodi alpha;
22      private Taistelu kappa;
23      private JButton a;
24      private JButton s;
25      private JButton d;
26  
27      public HyokkaysNappi(JTextArea kohde, Moodi alpha, String komento, Mobs enemy) {
28          this.hero = alpha.getHero();
29          this.kohde = kohde;
30          this.komento = komento;
31          this.alpha = alpha;
32          this.enemy = enemy;
33          this.kappa = kappa;
34          this.a = alpha.getA();
35          this.s = alpha.getS();
36          this.d = alpha.getD();
37      }
38  
39      @Override
40      public void actionPerformed(ActionEvent ae) {
41          this.komento = "A";
42          this.alpha.setKomento(komento);
43          this.alpha.getTeksti().setText(Gauntlet.action(enemy, hero, komento));
44          if (!enemy.tellAlive() || !hero.tellIfAlive()) {
45              if (hero.getCurrentHp() < 1) {
46                  ArrayList<JButton> k = new ArrayList();
47                  k.add(a);
48                  k.add(s);
49                  k.add(d);
50  
51                  for (JButton currentButton : k) {
52                      for (ActionListener al : currentButton.getActionListeners()) {
53                          currentButton.removeActionListener(al);
54                      }
55                  }
56                  alpha.getTeksti().setText("Gameover (Press D to restart(disabled))");
57                  //AloitusNappi e = new AloitusNappi(this.teksti, this.a, this.s, this.d, this.hero);
58                  //this.d.addActionListener(e);
59              } else if (enemy.getHp() < 1) {
60                  hero.gainXp(enemy.giveXp());
61                  alpha.getTeksti().setText("Gained " + enemy.giveXp() + " xp");
62                  if (hero.getSp() > 0) {
63                      alpha.getTeksti().setText("Enemy defeated: Unused skillpoints " + hero.getSp() + NL + "Agility gives small bonus to crit chance, hit chance and blocked damage "
64                              + NL + " Streight increases damage " + NL
65                              + " Hp gives you more durability, lose it all and you die.");
66                      ArrayList<JButton> k = new ArrayList();
67                      k.add(a);
68                      k.add(s);
69                      k.add(d);
70  
71                      for (JButton currentButton : k) {
72                          for (ActionListener al : currentButton.getActionListeners()) {
73                              currentButton.removeActionListener(al);
74                          }
75                      }
76                      StreightKehitysNappi str = new StreightKehitysNappi(alpha.getTeksti(), alpha);
77                      this.s.addActionListener(str);
78                      AgilityKehitysNappi agi = new AgilityKehitysNappi(alpha.getTeksti(), alpha);
79                      this.a.addActionListener(agi);
80                      HpKehitysNappi hp = new HpKehitysNappi(alpha.getTeksti(), alpha);
81                      this.d.addActionListener(hp);
82                  }
83  
84              }
85  
86          }
87      }
88  }
