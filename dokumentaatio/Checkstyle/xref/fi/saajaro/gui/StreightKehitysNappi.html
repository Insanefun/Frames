25      public StreightKehitysNappi(JTextArea kohde, Moodi alpha) {

28          this.alpha = alpha;
29      }
30  
31      @Override
32      public void actionPerformed(ActionEvent ae) {
33          hero.gainStreight();
34          this.kohde.setText(hero.toString() + NL + "Agility gives small bonus to crit chance, hit chance and blocked damage "
35                  + NL + " Streight increases damage " + NL
36                  + " Hp gives you more durability, lose it all and you die.");
37          if (this.hero.getSp() == 0 && this.hero.getCurrentLevel() == 1) {
38              this.kohde.setText("Battle! A = attack, S = study, D = defend");
39              Taistelu p = new Taistelu(this.alpha);
40              p.taisto();
41          } else if (this.hero.getSp() == 0) {
42              Taistelu p = new Taistelu(this.alpha);
43              p.taisto();
44          }

46      }
