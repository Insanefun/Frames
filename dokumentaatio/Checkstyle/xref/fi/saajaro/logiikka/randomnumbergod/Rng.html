5   public class Rng {
6   
7       /**
8        * Arvo jota suurenpi satunnaisluvun on oltava jotta palautettaisi false.
9        */
10      private int arvo;
11      /**
12       * suurin mahdollinen arvo jonka satunnaisluku voi saada.
13       */
14      private int suurin;
15  
16      /**
17       * Asettaa olion arvon arvo ja arvon suurin parametreinä saatuihin arvoihin.
18       *
19       * @param arvo asettaa olion arvon arvoksi saadun arvon
20       * @param suurin asettaa olion arvon suurin arvoksi saadun arvon
21       */
22      public Rng(int arvo, int suurin) {
23          this.arvo = arvo;
24          this.suurin = suurin;
25  
26      }
27  
28      /**
29       * Metodi joka arpoo satunnaisluvun joka on enintään olioon tallennettu
30       * suurin arvo ja tarkistaa onko se suurenpi kuin olioon tallennettu arvo
31       * arvo.
32       *
33       * @return palauttaa true jos arvo oli suurenpi kuin arvottu luku
34       */
35      public boolean randomNumber() {
36          Random rn = new Random();
37          int vertailu = rn.nextInt(this.suurin);
38          if (this.arvo > vertailu) {
39              return true;
40          } else {
41              return false;
42          }
43      }
44  }
