84      public void taisto() {

86          //for(ActionListener l : this.a.getActionListeners() ){
87          // k.add(l);
88          //}
89          ArrayList<JButton> k = new ArrayList();
90          k.add(a);
91          k.add(s);
92          k.add(d);
93  
94          for (JButton currentButton : k) {
95              for (ActionListener al : currentButton.getActionListeners()) {
96                  currentButton.removeActionListener(al);
97              }
98  
99          }
100         this.teksti.setText("Le battle");
101         Mobs enemy = Gauntlet.generateEnemy();
102         HyokkaysNappi y = new HyokkaysNappi(this.teksti, this, "A", enemy);
103         this.a.addActionListener(y);
104         battle(this);
105

108     public void battle(Moodi alpha) {
109         Pelaaja hero = alpha.getHero();
110         Mobs enemy = Gauntlet.generateEnemy();
111         enemy.modifyDodge(hero.hitBonus());
112         alpha.getTeksti().setText(enemy.encounter());
113         while (hero.tellIfAlive() && enemy.tellAlive()) {
114             alpha.getTeksti().setText(alpha.getTeksti().getText() + NL + "Choose a command: A = Attack, S = Tattle, D = Defend");
115             Gauntlet.turn(enemy, alpha);
116         }
117     }
