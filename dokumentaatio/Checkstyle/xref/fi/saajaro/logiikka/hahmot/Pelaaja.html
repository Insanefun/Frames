79      /**
80       * Nostaa olion xp arvoa parametrina saadun määrän ja jos asetettu raja
81       * ylitetään nostetaan olion level muuttujaa ja asetetaan sen xp nollaan.
82       *
83       * @param exp saatu xp
84       * @see fi.saajaro.logiikka.hahmot.Pelaaja#xp
85       * @see fi.saajaro.logiikka.hahmot.Pelaaja#level
86       */
87  
88      public void gainXp(int exp) {
89          this.xp = this.xp + exp;
90          if (this.xp >= this.level * 5 + 5) {
91              this.level = this.level + 1;
92              this.sp = this.sp + 3;
93              this.xp = 0;
94          }
95  
96      }

142     /**
143      * Laskee olion currentHp arvoa laskettuaan lopullisen damage arvon saadusta
144      * damage arvosta. Muutta olion alive arvoksi false jos laskee currentHp
145      * arvon alle yhteen.
146      *
147      * @param damage oletus arvoinen damage
148      * @see fi.saajaro.logiikka.hahmot.Pelaaja#currentHp
149      */
150 
151     public void takeDamage(int damage) {
152         int take = this.calcDamageTaken(damage);
153         if (take < 0) {
154             take = 0;
155         }
156         this.currentHp = this.currentHp - take;
157         this.toiminta = ("You took " + take + " damage from the attack of the enemy, " + this.currentHp + " hp left.");
158         if (this.currentHp < 1) {
159             this.alive = false;
160         }
161 
162     }

164     /**
165      * Olion metodi joka laskee otetun vaurion suuruden ja jos oliolla on
166      * bonusDefence arvo 1 laskee vaurion suuruutta.
167      *
168      * @param damage saatu damage arvo
169      * @return mahdollisesti muuttunut damage arvo
170      * @see fi.saajaro.logiikka.hahmot.Pelaaja#bonusDefence
171      */
172 
173     public int calcDamageTaken(int damage) {
174         if (this.bonusDefence == 1) {
175             damage = damage - this.agility / 3;
176             damage = damage - damage % 2;
177             this.bonusDefence = 0;
178         }
179         return damage;
180     }
181 
182     /**
183      * olion metodi jota kutsuttaessa saadaan selville sen damage arvo. Käyttää
184      * apunaan metodia calcDamage.
185      *
186      * @return laskettu damage arvo
187      * @see fi.saajaro.logiikka.hahmot.Pelaaja#calcDamage(int)
188      */
189 
190     public int dealDamage() {
191         int ignoredStr = this.streight % 2;
192         int strDmg = (this.streight - ignoredStr) / 2;
193         int damage = calcDamage(1 + strDmg);
194         return damage;
195     }

197     /**
198      * laskee parametrina saadun damagen loppu arvon käyttämällä apunaan
199      * metodeja crit() ja damageBonus().
200      *
201      * @param damage parametrina saatu damage jonka loppu arvo lasketaan
202      * @return damagen loppu arvo
203      * @see fi.saajaro.logiikka.hahmot.Pelaaja#crit(int)
204      * @see fi.saajaro.logiikka.hahmot.Pelaaja#damageBonus(int)
205      */
206 
207     public int calcDamage(int damage) {
208         damage = crit(damage);
209         damage = damageBonus(damage);
210 
211         return damage;
212     }

214     /**
215      * Metodi joka luo satunnaisluvun käyttäen Rng oliota ja muokkaa parametrinä
216      * saatua damage arvoa jos oliolle laskettava critchance on suurempi kuin
217      * arvottu satunnaisluku.
218      *
219      * @param damage parametrina saatu hyökkäyksen oletus damage arvo
220      * @return mahdollisesti muuttunut hyökkäyksen damage arvo
221      */
222 
223     public int crit(int damage) {
224         int critchance = this.agility * 3 - this.level - 2;
225         if (critchance < 0) {
226             critchance = 0;
227         }
228         Rng crit = new Rng(critchance, 100);
229         if (crit.randomNumber() == true) {
230             damage = 2 * damage;
231         }
232         return damage;
233     }

235     /**
236      * Metodi joka tarkistaa onko oliolla attackBonusta ja lisää parametrinä
237      * annettua damage arvoa riippuen attackBonus arvosta ja asettaa
238      * attackBonuksen nollaan.
239      *
240      * @param damage parametrina saatu hyökkäyksen oletus damage arvo
241      * @return mahdollisesti muuttunut damage arvo
242      */
243 
244     public int damageBonus(int damage) {
245         if (this.attackBonus == 1) {
246             int bonus = this.streight - this.streight % 2;
247             damage = damage + bonus / 2;
248             this.attackBonus = 0;
249         }
250         return damage;
251     }
